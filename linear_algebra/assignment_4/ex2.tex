\section*{Problem 2}

Find the matrix of the linear transformation:

\[
	T(x_1, x_2) = \genericmat{p}{2x_1 - x_2 \\ x_1 + x_2 \\ x_2}
\]

\[
	\begin{aligned}
		T(\vec{e_1}) &= T \genpmat{1\\0} = \genpmat{2\\1\\0} \\
		T(\vec{e_2}) &= T \genpmat{0\\1} = \genpmat{-1\\1\\1} \\
		A &= \genpmat{2&-1\\1&1\\0&1}
	\end{aligned}
\]
